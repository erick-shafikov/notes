# методы списка

```python
some_list = [1, 2, 3, 4, 5]
# поменяет список, добавить один элемент
some_list.append(100)
some_list.append([] * 3)  # вставить три элемента
# вставить на позицию 3 значение -1000 
some_list.insert(3, -1000)  # [1,2,3,-1000,4,5]
# удалить по значению (True удаляет 1, а False - 0)
some_list.remove(1)  # [2,3,4,5]
# удалить последний и вернуть его
last = some_list.pop()
# удалить по индексу
last = some_list.pop(3)
# очистить
some_list.clear()
# копия
copy = some_list.copy()
# методы поиска
# количество значение
some_list.count(1)  # 1
some_list.index(1)  # 0
# начиная со 2 найти 1
some_list.index(1, 2)  # 0
# поменять порядок
some_list.reverse()
# сортировка, меняет список от меньшего к большему
some_list.sort()
# сортировка от большего к меньшему
some_list.sort(reverse=True)
```
