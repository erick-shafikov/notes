# createRoot

- создает корневой элемент

```js
import { App } from "./app";
import { createRoot } from "react-dom/client";

const container = document.getElementById("root");

// This is the first step
const root = createRoot(container);

// Then, the second
root.render(<App />);
```

сам createRoot это функция, которая имеет сигнатуру

```ts
function createRoot(
  container: Element | Document | DocumentFragment,
  options?: CreateRootOptions
): RootType;

type CreateRootOptions = {
  unstable_strictMode: boolean; //StrictMode на корневом уровне
  unstable_concurrentUpdatesByDefault: boolean; //параллельное обновление
  unstable_transitionCallbacks: TransitionTracingCallbacks;
  identifierPrefix: string; //React Flight
  onRecoverableError: (error: any) => void; // при автоматическом восстановлении React после ошибок
};
```

в следствии чего, идет проверка на валидно элемента (body не лучшее место, так как может быть использовано другими)

```js
const createRoot = () => {
  //первый шаг
  //идет инициализация всех переданных параметров в createRoot
  let isStrictMode = false;
  let identifierPrefix = "";
  // другие опции

  if (options) {
    if (options.unstable_strictMode === true) {
      isStrictMode = true;
    }
    // ...
  }

  //создается fiberRoot - корневой fiber
  // createContainer - ниже
  const fiberRoot = createContainer(
    container,
    ConcurrentRoot,
    null,
    isStrictMode,
    isConcurrentUpdatesByDefault,
    identifierPrefix,
    onRecoverableError,
    transitionCallbacks
  );

  container.__reactContainer$randomValue = fiberRoot.current;
  Dispatcher.current = ReactDOMClientDispatcher;
};

const createContainer = (
  container,
  ConcurrentRoot,
  isStrictMode,
  isConcurrentUpdatesByDefault,
  identifierPrefix,
  onRecoverableError,
  transitionCallbacks
) => {
  // создается FiberRootNode
  const fiberRoot = new FiberRootNode(
    container, // the host element
    tag, // ConcurrentRoot
    hydrate, // false for this path
    identifierPrefix, // options?.identifierPrefix || ''
    onRecoverableError // options?.onRecoverableError || reportError || console.error
  );

  // далее создается FiberNode
  const unutilizedFiber = new FiberNode(
    HostRoot, // tag
    null, // pendingProps
    null, // key
    mode // deduced react mode (strict mode, strict effects, concurrent updates..)
  );

  //они ссылаются друг на друга
  fiberRoot.current = unitializedFiber;
  unitializedFiber.stateNode = fiberRoot;

  // инициализация состояния uninitializedFiber
  uninitializedFiber.memoizedState = {
    element: null, // initialChildren
    isDehydrated: false, // hydrate
    cache: null, // put behind a feature flag
  };

  uninitializedFiber.updateQueue = {
    baseState: fiber.memoizedState, // we just created this above
    firstBaseUpdate: null,
    lastBaseUpdate: null,
    shared: {
      pending: null,
      lanes: NoLanes, // 0
      hiddenCallbacks: null,
    },
    callbacks: null,
  };

  // возвращает fiberRoot
  return fiberRoot;
};
```

```tsx
export function updateContainer(
  element: ReactNodeList, // children
  container: OpaqueRoot, // OpaqueRoot = FiberRoot = new FiberRootNode
  parentComponent?: React$Component<any, any>,
  callback?: Function
): Lane {
  //FiberRootNode или FiberNode
  const current = container.current;
  // запрос полосы обновления
  // вызов setState во время рендеринга возвращается полоса с наивысшим приоритетом :
  const lane = requestUpdateLane(current);

  const update = createUpdate(lane);
  update.payload = { element };
  update.callback = callback;

  //обновления в очередь
  const root = enqueueUpdate(current, update, lane);
  scheduleUpdateOnFiber(root, current, lane);
  entangleTransitions(root, current, lane);

  // создается обновление для дорожки
  const update = {
    lane,
    tag: UpdateState, // 0

    callback, // callback or null
    payload: { element }, // element are the root children

    next: null,
  };

  // добавляем обновление в Fiber
  /* enqueueUpdate пройдет следующие этапы:
  - вернет null если fiber.updateQueueравен нулю
  - предупредить о разработке о вложенных setStateвызовах
  - обработка случая классовых компонентов
  - обработка функционального компонента с помощью enqueueConcurrentClassUpdate
  */
  // current: FiberNode
  // update: Update
  // lane: number (update lane)
  const root: FiberRoot | null = enqueueUpdate(current, update, lane);

  // планировка обновления fiber
  scheduleUpdateOnFiber(fiberRoot, currentFiber, updateLane);

  // добавляется fiber
}
```

```ts
function scheduleTaskForRootDuringMicrotask(
  root: FiberRoot,
  currentTime: number
): Lane {
  const pendingLanes = root.pendingLanes;
  const suspendedLanes = root.suspendedLanes;
  const pingedLanes = root.pingedLanes;
  const expirationTimes = root.expirationTimes;

  let lanes = pendingLanes & ~RetryLanes;
  while (lanes > 0) {
    // in the following two lines, we will get the index of the highest priority
    // lane from the lanes, and then we will shift 1 by that index so we obtain
    // the current lane with only one significant bit (power of 2)
    const index = pickArbitraryLaneIndex(lanes);
    const lane = 1 << index;

    // the expiration time for this lane
    const expirationTime = root.expirationTimes[index];

    // this means it is not scheduled yet
    if (expirationTime !== NoTimestamp) {
      // this means this lane isn't suspended or is pinged
      // pinged lanes are the lanes that were suspending the root when
      // the suspending promise did resolve.
      if (
        (lane & suspendedLanes) === NoLanes ||
        (lane & pingedLanes) !== NoLanes
      ) {
        expirationTimes[index] = computeExpirationTime(lane, currentTime);
      }
    } else if (expirationTime <= currentTime) {
      // this is when it is not expired
      root.expiredLanes |= lane;
    }

    lanes &= ~lane;
  }
}

const nextLanes = getNextLanes(root, NoLanes);
```
